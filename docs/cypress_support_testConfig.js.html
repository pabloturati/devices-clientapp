/**
 * Navigates the robot to the given route
 * @function
 * @param {*} route
 * @param {*} verifyCallback
 */

export const navigateTo = (route, verifyCallback) => {
  cy.visit(route)
  cy.url().should('include', route)
  if (verifyCallback) verifyCallback()
}
